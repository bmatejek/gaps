2dview: visualize an h5 volume slice by slice with value query
	2dview [-v] filename dataset
	
3dview: visualize an h5 volume with 3d renderings
	3dview [-v] filename dataset

downsampleview: visualize the skeletons as lower resolutions
	downsampleview [-v] prefix

overlay: overlay a label volume over an image volume
	overlay [-v] image_filename image_dataset label_filename label_dataset

skeletonview: visualize skeletons as normal resolution
	skeletonview [-v] prefix
